### 三、文件向量化处理

#### 3.1 向量化基本原理
```python
class Vectorizer:
    def __init__(self, model_name='text-embedding-ada-002'):
        self.model = self._load_model(model_name)
        self.chunk_size = 1000  # 文本分块大小
        self.overlap = 100     # 分块重叠
        
    def vectorize_file(self, file_path, file_type):
        """
        向量化文件的核心方法
        """
        # 1. 根据文件类型选择处理方式
        if file_type == 'txt':
            return self._vectorize_text(file_path)
        elif file_type == 'pdf':
            return self._vectorize_pdf(file_path)
        elif file_type in ['jpg', 'png']:
            return self._vectorize_image(file_path)
        else:
            raise ValueError(f"不支持的文件类型: {file_type}")
            
    def _vectorize_text(self, file_path):
        """
        文本文件向量化
        """
        # 1. 读取文本
        with open(file_path, 'r', encoding='utf-8') as f:
            text = f.read()
            
        # 2. 文本分块
        chunks = self._chunk_text(text)
        
        # 3. 向量化每个分块
        vectors = []
        for i, chunk in enumerate(chunks):
            vector = self.model.encode(chunk)
            vectors.append({
                'chunk_id': f"chunk_{i}",
                'text': chunk,
                'vector': vector,
                'metadata': {
                    'start_pos': i * (self.chunk_size - self.overlap),
                    'end_pos': min((i + 1) * self.chunk_size - self.overlap, len(text))
                }
            })
            
        return vectors
```

```python
        """
        PDF文件向量化

        text = self._extract_pdf_text(file_path)
        
        # 2. 提取PDF元数据

        images = self._extract_images(file_path)
        
        # 4. 结构化处理
        structured_data = self._structure_pdf_content(text, tables, images)
        
        # 5. 向量化结构化内容
        vectors = []
        for section in structured_data:
            vector = self.model.encode(section['content'])
            vectors.append({
                'section_id': section['id'],
                'content': section['content'],
                'vector': vector,
                'metadata': {
                    'page': section['page'],
                    'type': section['type'],
                    **metadata
                }
            })
            
        return vectors
```

#### 4.2 向量检索机制
```python
class VectorRetriever:
    def __init__(self, vector_store, similarity_threshold=0.7):
        self.vector_store = vector_store
        self.threshold = similarity_threshold
        
    def retrieve_similar(self, query_vector, top_k=5, filters=None):
        """
        检索相似向量
        """
        # 1. 执行相似度搜索
        results = self.vector_store.query(
            query_embeddings=[query_vector],
            n_results=top_k * 2,  # 获取更多结果用于过滤
            where=filters
        )
        
        # 2. 过滤低相似度结果
        filtered_results = []
        for i, distance in enumerate(results['distances'][0]):
            similarity = 1 - distance  # 转换为相似度
            
            if similarity >= self.threshold:
                filtered_results.append({
                    'id': results['ids'][0][i],
                    'similarity': similarity,
                    'metadata': results['metadatas'][0][i],
                    'distance': distance
                })
                
        # 3. 按相似度排序并返回top_k
        filtered_results.sort(key=lambda x: x['similarity'], reverse=True)
        return filtered_results[:top_k]
```

#### 4.3 混合检索策略
```python
class HybridRetriever:
    def __init__(self, vector_retriever, keyword_retriever):
        self.vector_retriever = vector_retriever
        self.keyword_retriever = keyword_retriever
        
    def hybrid_retrieve(self, query, query_vector=None, top_k=5):
        """
        混合检索：向量检索 + 关键词检索
        """
        # 1. 向量检索
        if query_vector:
            vector_results = self.vector_retriever.retrieve_similar(
                query_vector, top_k=top_k
            )
        else:
            vector_results = []
            
        # 2. 关键词检索
        keyword_results = self.keyword_retriever.retrieve(
            query, top_k=top_k
        )
        
        # 3. 结果融合（加权评分）
        all_results = {}
        
        # 处理向量检索结果
        for result in vector_results:
            score = result['similarity'] * 0.7  # 向量检索权重
            all_results[result['id']] = {
                'score': score,
                'source': 'vector',
                **result
            }
            
        # 处理关键词检索结果
        for result in keyword_results:
            if result['id'] in all_results:
                # 如果已经存在，增加分数
                all_results[result['id']]['score'] += result['relevance'] * 0.3
                all_results[result['id']]['source'] = 'hybrid'
            else:
                all_results[result['id']] = {
                    'score': result['relevance'] * 0.3,  # 关键词检索权重
                    'source': 'keyword',
                    **result
                }
                
        # 4. 按总分排序
        sorted_results = sorted(
            all_results.values(),
            key=lambda x: x['score'],
            reverse=True
        )
        
        return sorted_results[:top_k]
```

```python
class MultimodalFileSearch:

        elif query_type == 'hybrid':
            # 混合搜索
            text_results = self.search_files(query, 'text')
            image_results = self.search_files(query, 'image')
            results = self._merge_results(text_results, image_results)
```

```python
class ErrorHandler:
    def __init__(self):
        self.retry_strategy = ExponentialBackoffRetry()
        
    def handle_vectorization_error(self, file_path, error):
        """
        处理向量化错误
        """
        error_type = type(error).__name__
        
        if error_type == 'ModelError':
            # 模型错误：尝试使用备用模型
            return self._try_alternative_model(file_path)
            
        elif error_type == 'FileFormatError':
            # 文件格式错误：尝试转换格式
            return self._convert_file_format(file_path)
            
        elif error_type == 'MemoryError':
            # 内存错误：分块处理
            return self._process_in_chunks(file_path)
            
        elif error_type == 'TimeoutError':
            # 超时错误：重试
            return self.retry_strategy.retry(
                lambda: self._vectorize_file(file_path)
            )
            
        else:
            # 未知错误：记录并跳过
            logger.error(f"无法处理的错误: {error}")
```

```python
def vectorize_text_file(file_path):
    with open(file_path, 'r') as f:
        text = f.read()
    
    # 简单分块
    chunks = [text[i:i+1000] for i in range(0, len(text), 1000)]
    
    # 向量化
    embeddings = model.encode(chunks)
    
    # 存储
    for i, (chunk, embedding) in enumerate(zip(chunks, embeddings)):
        collection.add(
            embeddings=[embedding.tolist()],
            documents=[chunk],
            metadatas=[{"file": file_path, "chunk": i}],
            ids=[f"{file_path}_chunk_{i}"]
        )
    
    return len(chunks)

# 使用
vectorize_text_file("example.txt")
```
