### 1.1 集中式MAS（昨天分层架构的扩展）

#### 核心概念
**"一个中心Agent指挥多个工作Agent"**

#### 架构图示
```
        ┌─────────────────┐
        │   中心Agent     │ ← 昨天分层架构的协调层扩展
        │  (Coordinator)  │
        └────────┬────────┘
         ┌───────┴───────┐
    ┌────┴────┐   ┌─────┴─────┐
    │  Agent1 │   │   Agent2  │ ← 多个执行层Agent
    └─────────┘   └───────────┘
```

#### 伪代码示例（理解架构思想）
```python
# 集中式MAS伪代码示例 - 理解架构思想
class CentralizedMAS:
    def __init__(self):
        # 中心Agent（协调者）
        self.coordinator = CoordinatorAgent()  # 中心协调Agent
        
        # 工作Agent集合
        self.workers = [
            SearchAgent(),      # 搜索Agent
            AnalysisAgent(),    # 分析Agent
            ReportAgent()       # 报告Agent
        ]
    
    def process_task(self, task):
        # 1. 中心Agent接收任务
        print("中心Agent接收任务...")  # 日志输出
        
        # 2. 中心Agent分析任务并分配
        assignments = self.coordinator.analyze_and_assign(task, self.workers)  # 任务分析分配
        
        # 3. 工作Agent并行执行
        results = []
        for worker, subtask in assignments:  # 遍历分配的任务
            result = worker.execute(subtask)  # 工作Agent执行子任务
            results.append(result)            # 收集结果
        
        # 4. 中心Agent整合结果
        final_result = self.coordinator.integrate(results)  # 结果整合
        
        return final_result  # 返回最终结果
```

### 1.2 分布式MAS（昨天黑板架构的扩展）

#### 核心概念
**"多个平等Agent自主协作，没有中心控制"**

#### 架构图示
```
    ┌─────────┐   ┌─────────┐   ┌─────────┐
    │ Agent A │   │ Agent B │   │ Agent C │ ← 平等Agent
    └────┬────┘   └────┬────┘   └────┬────┘
         │             │             │
    ┌────┴─────────────┴─────────────┴────┐
    │          通信网络/消息总线           │ ← 昨天黑板架构的通信扩展
    └─────────────────────────────────────┘
```

#### 伪代码示例（理解架构思想）
```python
# 分布式MAS伪代码示例 - 理解架构思想
class DistributedMAS:
    def __init__(self):
        # 平等Agent集合（没有中心控制）
        self.agents = [
            SensorAgent(),      # 传感器Agent
            ProcessorAgent(),   # 处理器Agent
            ActuatorAgent()     # 执行器Agent
        ]
        
        # 消息总线（Agent间通信）
        self.message_bus = MessageBus()  # 消息总线
    
    def start_system(self):
        # 1. 所有Agent启动并注册到消息总线
        for agent in self.agents:        # 遍历所有Agent
            agent.register(self.message_bus)  # Agent注册到消息总线
        
        # 2. Agent通过消息总线自主通信协作
        # 没有中心控制，Agent根据消息自主决策
        print("分布式MAS启动：Agent自主协作中...")  # 启动日志
```

        self.home_agents["climate"].add_dependency("occupancy")  # 添加占用依赖
        
        # 照明控制Agent需要知道时间状态和占用状态

        self.home_agents["lighting"].add_dependency("occupancy")    # 添加占用依赖
        
        # 安防监控Agent需要知道占用状态
        self.home_agents["security"].add_dependency("occupancy")  # 添加占用依赖

    def update_home_state(self, updates: Dict[str, Any]):
        """更新家居状态，触发相关Agent响应"""
        print(f"[状态更新] 更新家居状态: {updates}")  # 状态更新日志
        
        # 记录事件
        event = {
            "timestamp": datetime.now().isoformat(),  # 时间戳
            "event_type": "state_update",             # 事件类型
            "updates": updates,                       # 更新内容
            "previous_state": self.home_state.copy()  # 先前状态
        }
        self.event_log.append(event)  # 添加到事件日志
        
        # 更新状态
        self.home_state.update(updates)  # 更新状态
        
        # 广播状态更新消息
        self.message_bus.broadcast({
            "type": "state_update",      # 消息类型
            "data": updates,             # 更新数据
            "timestamp": datetime.now().isoformat()  # 时间戳
        })
        
        # 触发相关Agent的响应
        self._trigger_agent_responses(updates)  # 触发Agent响应

        """根据状态更新触发相关Agent响应"""
        updated_keys = updates.keys()  # 获取更新的键
        
        # 检查每个Agent的依赖是否被更新
        for agent_name, agent in self.home_agents.items():
            dependencies = agent.get_dependencies()  # 获取Agent依赖
            
            # 如果Agent的依赖被更新，则触发Agent处理
            if any(dep in updated_keys for dep in dependencies):
                print(f"[触发响应] {agent_name} Agent 依赖已更新，触发处理")  # 触发日志
                agent.process_state_update(updates, self.home_state)  # 处理状态更新

class MessageBus:
    """消息总线：Agent间通信基础设施"""
    
    def __init__(self):
        self.subscribers = {}  # 订阅者字典 {agent_name: callback}
        self.message_history = []  # 消息历史
        self.message_count = 0  # 消息计数
    
    def register(self, agent_name: str, callback):
        """注册Agent到消息总线"""
        self.subscribers[agent_name] = callback  # 存储回调函数
        print(f"[消息总线] {agent_name} 已注册")  # 注册日志
    
    def broadcast(self, message: Dict[str, Any]):
        """广播消息给所有订阅者"""
        self.message_count += 1  # 消息计数增加
        
        # 记录消息历史
        message_record = {
            "id": self.message_count,          # 消息ID
            "timestamp": datetime.now().isoformat(),  # 时间戳
            "type": message.get("type", "unknown"),   # 消息类型
            "data": message.get("data", {}),          # 消息数据
            "recipients": list(self.subscribers.keys())  # 接收者
        }
        self.message_history.append(message_record)  # 添加到历史
        
        print(f"[消息总线] 广播消息 #{self.message_count}: {message['type']}")  # 广播日志
        
        # 发送给所有订阅者
        for agent_name, callback in self.subscribers.items():
            try:
                callback(message)  # 调用回调函数
            except Exception as e:
                print(f"[消息总线] 发送给 {agent_name} 失败: {str(e)}")  # 错误日志
    
    def send_to(self, recipient: str, message: Dict[str, Any]):
        """发送消息给特定Agent"""
        if recipient in self.subscribers:
            self.message_count += 1  # 消息计数增加
            
            # 记录消息历史
            message_record = {
                "id": self.message_count,          # 消息ID
                "timestamp": datetime.now().isoformat(),  # 时间戳
                "type": message.get("type", "unknown"),   # 消息类型
                "data": message.get("data", {}),          # 消息数据
                "recipients": [recipient]                 # 接收者
            }
            self.message_history.append(message_record)  # 添加到历史
            
            print(f"[消息总线] 发送消息 #{self.message_count} 给 {recipient}: {message['type']}")  # 发送日志
            
            try:
                self.subscribers[recipient](message)  # 调用接收者的回调
            except Exception as e:
                print(f"[消息总线] 发送给 {recipient} 失败: {str(e)}")  # 错误日志
        else:
            print(f"[消息总线] 未知接收者: {recipient}")  # 未知接收者警告
    
    def get_stats(self) -> Dict[str, Any]:
        """获取消息总线统计"""
        return {
            "total_messages": self.message_count,  # 总消息数
            "subscriber_count": len(self.subscribers),  # 订阅者数量
            "recent_messages": self.message_history[-3:] if self.message_history else []  # 最近3条消息
        }

class BaseHomeAgent:
    """家居Agent基类"""
    
    def __init__(self, name: str):
        self.name = name  # Agent名称
        self.dependencies = []  # 依赖的状态键
        self.action_history = []  # 动作历史
    
    def register(self, message_bus: MessageBus):
        """注册到消息总线"""
        self.message_bus = message_bus  # 存储消息总线引用
        self.message_bus.register(self.name, self.receive_message)  # 注册回调函数
    
    def add_dependency(self, state_key: str):
        """添加状态依赖"""
        if state_key not in self.dependencies:
            self.dependencies.append(state_key)  # 添加依赖
    
    def get_dependencies(self) -> List[str]:
        """获取依赖列表"""
        return self.dependencies.copy()  # 返回依赖副本
    
    def receive_message(self, message: Dict[str, Any]):
        """接收消息（由消息总线调用）"""
        print(f"[{self.name}] 收到消息: {message.get('type', 'unknown')}")  # 接收消息日志
        
        # 记录消息
        self.action_history.append({
            "timestamp": datetime.now().isoformat(),  # 时间戳
            "action": "receive_message",              # 动作类型
            "message_type": message.get("type"),      # 消息类型
            "message_data": message.get("data", {})   # 消息数据
        })
    
    def process_state_update(self, updates: Dict[str, Any], current_state: Dict[str, Any]):
        """处理状态更新（由系统调用）"""
        raise NotImplementedError("子类必须实现此方法")  # 抽象方法
    
    def take_action(self, action: str, details: Dict[str, Any]):
        """执行动作并记录"""
        print(f"[{self.name}] 执行动作: {action} - {details}")  # 动作执行日志
        
        # 记录动作
        self.action_history.append({
            "timestamp": datetime.now().isoformat(),  # 时间戳
            "action": action,                         # 动作类型
            "details": details                        # 动作详情
        })
        
        # 可以通过消息总线通知其他Agent
        if hasattr(self, 'message_bus'):
            self.message_bus.broadcast({
                "type": "agent_action",      # 消息类型
                "agent": self.name,          # Agent名称
                "action": action,            # 动作类型
                "details": details,          # 动作详情
                "timestamp": datetime.now().isoformat()  # 时间戳
            })

        if "temperature" in updates:
            new_temp = updates["temperature"]  # 新温度
            self._adjust_climate(new_temp, current_state.get("occupancy"))  # 调整气候
        
        # 检查占用状态更新
        if "occupancy" in updates:
            occupancy = updates["occupancy"]  # 占用状态
            self._adjust_for_occupancy(occupancy)  # 根据占用调整
    
    def _adjust_climate(self, current_temp: float, occupancy: str):
        """根据当前温度调整气候"""
        temp_diff = current_temp - self.target_temperature  # 温度差
        
        if abs(temp_diff) > 2.0:  # 如果温差超过2度
            if temp_diff > 0:  # 太热
                action = "启动空调" if occupancy == "home" else "节能模式降温"  # 根据占用选择动作
                self.take_action("adjust_temperature", {  # 执行温度调整
                    "action": action,
                    "current_temp": current_temp,
                    "target_temp": self.target_temperature,
                    "difference": temp_diff
                })
            else:  # 太冷
                action = "启动暖气" if occupancy == "home" else "节能模式保温"  # 根据占用选择动作
                self.take_action("adjust_temperature", {  # 执行温度调整
                    "action": action,
                    "current_temp": current_temp,
                    "target_temp": self.target_temperature,
                    "difference": temp_diff
                })
    
    def _adjust_for_occupancy(self, occupancy: str):
        """根据占用状态调整"""
        if occupancy == "away":  # 如果外出
            self.target_temperature = 18.0  # 节能模式目标温度
            self.take_action("set_energy_saving", {  # 执行节能设置
                "reason": "住户外出",
                "new_target_temp": self.target_temperature
            })
        elif occupancy == "home":  # 如果在家

### 1.3 混合式MAS（结合集中式和分布式优点）

#### 核心概念
**"既有中心协调，又有Agent自主性"**

#### 架构图示
```
        ┌─────────────────┐
        │   中心协调层     │ ← 集中式部分
        │  (Coordinator)  │
        └────────┬────────┘
         ┌───────┴───────┐
    ┌────┴────┐   ┌─────┴─────┐
    │ 集群A   │   │  集群B    │ ← 分布式集群
    │ (自治)  │   │  (自治)   │
    └────┬────┘   └─────┬─────┘
    ┌────┴────┐   ┌─────┴─────┐
    │ Agent A1│   │ Agent B1  │ ← 集群内Agent
    │ Agent A2│   │ Agent B2  │
    └─────────┘   └───────────┘
```

#### 伪代码示例（理解架构思想）
```python
# 混合式MAS伪代码示例 - 理解架构思想
class HybridMAS:
    def __init__(self):
        # 中心协调层
        self.coordinator = CentralCoordinator()  # 中心协调器
        
        # 分布式集群
        self.clusters = {
            "data_cluster": DataProcessingCluster(),   # 数据处理集群
            "ui_cluster": UserInterfaceCluster(),      # 用户界面集群
            "api_cluster": APIIntegrationCluster()     # API集成集群
        }
    
    def handle_complex_task(self, task):
        # 1. 中心协调层接收任务并分析
        analysis = self.coordinator.analyze_task(task)  # 任务分析
        
        # 2. 中心协调层分配任务到集群
        cluster_assignments = self.coordinator.assign_to_clusters(analysis)  # 集群分配
        
        # 3. 集群内部自主协作（分布式）
        cluster_results = {}
        for cluster_name, subtask in cluster_assignments.items():  # 遍历集群分配
            cluster = self.clusters[cluster_name]                  # 获取集群
            result = cluster.process_autonomously(subtask)         # 集群自主处理
            cluster_results[cluster_name] = result                 # 收集结果
        
        # 4. 中心协调层整合结果
        final_result = self.coordinator.integrate_results(cluster_results)  # 结果整合
        
        return final_result  # 返回最终结果
```

class RecommendationOrchestrator:
    """推荐协调器：中心协调层"""
    
    def __init__(self):
        self.name = "推荐协调器"  # 协调器名称
        self.strategy_history = []  # 策略历史
    
    def analyze_user_context(self, user_id: str, user_profile: Dict[str, Any], 
                            interaction_history: List[Dict[str, Any]], context: Dict[str, Any]) -> Dict[str, Any]:
        """分析用户上下文，确定推荐策略"""
        print(f"[{self.name}] 分析用户 {user_id} 上下文...")  # 分析开始日志
        
        # 分析用户画像
        user_experience = self._analyze_user_experience(user_profile, interaction_history)  # 分析用户经验
        
        # 分析上下文
        context_type = self._analyze_context(context)  # 分析上下文类型
        
        # 确定推荐策略
        strategy = self._determine_strategy(user_experience, context_type)  # 确定策略
        
        # 记录策略决策
        strategy_record = {
            "timestamp": datetime.now().isoformat(),  # 时间戳
            "user_id": user_id,                       # 用户ID
            "user_experience": user_experience,       # 用户经验
            "context_type": context_type,             # 上下文类型
            "strategy": strategy,                     # 策略
            "context_details": context                # 上下文详情
        }
        self.strategy_history.append(strategy_record)  # 添加到历史
        
        return {
            "user_experience": user_experience,
            "context_type": context_type,
            "strategy": strategy,
            "analysis_timestamp": datetime.now().isoformat()
        }
    
    def _analyze_user_experience(self, user_profile: Dict[str, Any], interaction_history: List[Dict[str, Any]]) -> str:
        """分析用户经验水平"""
        interaction_count = len(interaction_history)  # 交互次数
        
        if interaction_count == 0:  # 新用户
            return "new_user"
        elif interaction_count < 10:  # 轻度用户
            return "light_user"
        elif interaction_count < 50:  # 中度用户
            return "medium_user"
        else:  # 重度用户
            return "heavy_user"
    
    def _analyze_context(self, context: Dict[str, Any]) -> str:
        """分析上下文类型"""
        # 简单上下文分析
        if context.get("time_of_day") in ["morning", "night"]:  # 早晨或夜晚
            return "focused_browsing"
        elif context.get("device") == "mobile":  # 移动设备
            return "quick_browsing"
        elif context.get("intent") == "purchase":  # 购买意图
            return "purchase_intent"
        else:  # 一般浏览
            return "general_browsing"
    
    def _determine_strategy(self, user_experience: str, context_type: str) -> str:
        """确定推荐策略"""
        # 策略决策矩阵
        strategy_matrix = {
            ("new_user", "general_browsing"): "explore_new_user",
            ("new_user", "purchase_intent"): "assist_purchase",
            ("light_user", "quick_browsing"): "quick_recommendations",
            ("medium_user", "focused_browsing"): "personalized_deep",
            ("heavy_user", "general_browsing"): "advanced_personalization",
            ("heavy_user", "purchase_intent"): "precision_recommendation"
        }
        
        # 获取策略，如果没有匹配则使用默认
        strategy = strategy_matrix.get((user_experience, context_type), "balanced_recommendation")  # 获取策略
        
        print(f"[{self.name}] 策略决策: {user_experience} + {context_type} → {strategy}")  # 策略决策日志
        
        return strategy
    
    def select_clusters(self, analysis: Dict[str, Any]) -> List[str]:
        """根据分析结果选择推荐集群"""
        strategy = analysis.get("strategy", "balanced_recommendation")  # 获取策略
        
        # 集群选择映射
        cluster_selection = {
            "explore_new_user": ["collaborative", "content_based"],  # 探索新用户：协同+内容
            "assist_purchase": ["content_based", "context_aware"],   # 辅助购买：内容+上下文
            "quick_recommendations": ["collaborative", "hybrid"],    # 快速推荐：协同+混合
            "personalized_deep": ["content_based", "context_aware", "hybrid"],  # 深度个性化：内容+上下文+混合
            "advanced_personalization": ["collaborative", "content_based", "context_aware", "hybrid"],  # 高级个性化：全部
            "precision_recommendation": ["hybrid", "context_aware"],  # 精准推荐：混合+上下文
            "balanced_recommendation": ["collaborative", "content_based"]  # 平衡推荐：协同+内容
        }
        
        selected = cluster_selection.get(strategy, ["collaborative", "content_based"])  # 获取选择的集群
        
        return selected  # 返回选择的集群
    
    def integrate_and_rank(self, cluster_recommendations: Dict[str, List[Dict[str, Any]]], 
                          analysis: Dict[str, Any]) -> List[Dict[str, Any]]:
        """整合和排序集群推荐"""
        print(f"[{self.name}] 整合 {len(cluster_recommendations)} 个集群的推荐...")  # 整合开始日志
        
        # 收集所有推荐
        all_recommendations = []
        for cluster_name, recommendations in cluster_recommendations.items():
            for rec in recommendations:
                rec["source_cluster"] = cluster_name  # 添加来源集群
                all_recommendations.append(rec)  # 添加到总列表
        
        print(f"[{self.name}] 总共收集到 {len(all_recommendations)} 个推荐")  # 收集日志
        
        # 去重（基于商品ID）
        unique_recommendations = {}
        for rec in all_recommendations:
            item_id = rec.get("item_id")  # 获取商品ID
            if item_id:
                if item_id not in unique_recommendations:
                    unique_recommendations[item_id] = rec  # 存储唯一推荐
                else:
                    # 合并来源集群
                    existing = unique_recommendations[item_id]  # 现有推荐
                    if "source_clusters" not in existing:
                        existing["source_clusters"] = [existing.get("source_cluster")]  # 初始化来源集群列表
                    existing["source_clusters"].append(rec.get("source_cluster"))  # 添加来源集群
                    existing["confidence"] = max(existing.get("confidence", 0), rec.get("confidence", 0))  # 取最高置信度
        
        # 转换为列表
        unique_list = list(unique_recommendations.values())  # 转换为列表
        
        # 根据策略排序
        strategy = analysis.get("strategy", "balanced_recommendation")  # 获取策略
        
        if strategy in ["precision_recommendation", "assist_purchase"]:  # 精准推荐或辅助购买
            # 按置信度排序
            sorted_recommendations = sorted(unique_list, key=lambda x: x.get("confidence", 0), reverse=True)  # 按置信度降序
        elif strategy == "explore_new_user":  # 探索新用户
            # 按多样性排序（模拟）
            sorted_recommendations = self._diversify_recommendations(unique_list)  # 多样化排序
        else:  # 其他策略
            # 默认按置信度排序
            sorted_recommendations = sorted(unique_list, key=lambda x: x.get("confidence", 0), reverse=True)  # 按置信度降序
        
        # 限制数量
        final_recommendations = sorted_recommendations[:10]  # 取前10个
        
        print(f"[{self.name}] 最终生成 {len(final_recommendations)} 个排序后的推荐")  # 最终结果日志
        
        return final_recommendations  # 返回最终推荐
    
    def _diversify_recommendations(self, recommendations: List[Dict[str, Any]]) -> List[Dict[str, Any]]:
        """多样化推荐（简化实现）"""
        # 简单实现：随机打乱
        import random
        shuffled = recommendations.copy()  # 复制列表
        random.shuffle(shuffled)  # 随机打乱
        return shuffled  # 返回打乱后的列表

class BaseRecommendationCluster:
    """推荐集群基类"""

                                interaction_history: List[Dict[str, Any]], 
                                context: Dict[str, Any], analysis: Dict[str, Any]) -> List[Dict[str, Any]]:
        """生成推荐（需要子类实现）"""

    def _create_recommendation(self, item_id: str, item_name: str, confidence: float, 
                              reason: str) -> Dict[str, Any]:
        """创建推荐对象"""

        }

class CollaborativeFilteringCluster(BaseRecommendationCluster):
    """协同过滤集群"""
    
    def __init__(self):
        super().__init__("协同过滤集群")  # 调用父类初始化
        # 模拟的用户-商品交互矩阵（实际中会从数据库加载）
        self.user_item_matrix = self._create_sample_matrix()  # 创建样本矩阵
    
    def _create_sample_matrix(self) -> Dict[str, Dict[str, float]]:
        """创建样本用户-商品交互矩阵"""
        # 模拟数据
        return {
            "user1": {"item1": 5.0, "item2": 3.0, "item3": 4.5, "item5": 2.0},
            "user2": {"item1": 4.0, "item3": 5.0, "item4": 3.5, "item6": 4.0},
            "user3": {"item2": 4.5, "item4": 4.0, "item5": 3.0, "item7": 5.0},
            "user4": {"item1": 3.0, "item3": 4.0, "item6": 4.5, "item8": 3.5},
            "user5": {"item2": 5.0, "item5": 4.0, "item7": 3.5, "item9": 4.0}
        }
    
    def generate_recommendations(self, user_id: str, user_profile: Dict[str, Any], 
                                interaction_history: List[Dict[str, Any]], 
                                context: Dict[str, Any], analysis: Dict[str, Any]) -> List[Dict[str, Any]]:
        """生成协同过滤推荐"""
        print(f"[{self.name}] 为用户 {user_id} 生成协同过滤推荐...")  # 开始日志
        
        # 模拟协同过滤算法
        recommendations = []
        
        # 查找相似用户（简化实现）
        similar_users = self._find_similar_users(user_id)  # 查找相似用户
        
        # 从相似用户喜欢的商品中推荐
        for sim_user, similarity in similar_users[:3]:  # 取前3个相似用户
            if sim_user in self.user_item_matrix:
                user_items = self.user_item_matrix[sim_user]  # 相似用户的商品
                
                for item_id, rating in user_items.items():
                    # 计算推荐置信度
                    confidence = similarity * (rating / 5.0)  # 基于相似度和评分
                    
                    if confidence > 0.3:  # 置信度阈值
                        recommendation = self._create_recommendation(  # 创建推荐
                            item_id=item_id,
                            item_name=f"商品{item_id[-1]}",
                            confidence=confidence,
                            reason=f"相似用户{sim_user}喜欢此商品（评分:{rating}）"
                        )
                        recommendations.append(recommendation)  # 添加到推荐列表
        
        print(f"[{self.name}] 生成 {len(recommendations)} 个协同过滤推荐")  # 完成日志
        
        return recommendations[:5]  # 返回前5个推荐
    
    def _find_similar_users(self, target_user: str) -> List[tuple]:
        """查找相似用户（简化实现）"""
        # 模拟相似度计算
        similarities = []
        
        for user in self.user_item_matrix:
            if user != target_user:  # 排除目标用户自己
                # 简单相似度计算（实际中会用余弦相似度等）
                similarity = random.uniform(0.3, 0.9)  # 随机相似度
                similarities.append((user, similarity))  # 添加到列表
        
        # 按相似度排序
        similarities.sort(key=lambda x: x[1], reverse=True)  # 按相似度降序
        
        return similarities  # 返回相似用户列表

    def _analyze_user_preferences(self, interaction_history: List[Dict[str, Any]]) -> Dict[str, Any]:
        """分析用户偏好"""
        if not interaction_history:  # 如果没有交互历史
            return {"category": "electronics", "price_range": "medium"}  # 返回默认偏好
        
        # 简单分析：统计最常交互的商品类别
        category_counts = {}  # 类别计数
        for interaction in interaction_history:
            item_id = interaction.get("item_id")  # 获取商品ID
            if item_id and item_id in self.item_features:
                category = self.item_features[item_id].get("category")  # 获取类别
                if category:
                    category_counts[category] = category_counts.get(category, 0) + 1  # 增加计数
        
        # 找出最受欢迎的类别
        if category_counts:
            preferred_category = max(category_counts.items(), key=lambda x: x[1])[0]  # 最受欢迎的类别
        else:
            preferred_category = "electronics"  # 默认类别
        
        return {
            "preferred_category": preferred_category,  # 偏好类别
            "price_range": "medium",  # 价格范围
            "tags": []  # 标签
        }
    
    def _calculate_match_score(self, item_features: Dict[str, Any], user_preferences: Dict[str, Any]) -> float:
        """计算商品与用户偏好的匹配度"""
        score = 0.0  # 初始分数
        
        # 类别匹配
        if item_features.get("category") == user_preferences.get("preferred_category"):  # 类别匹配
            score += 0.5  # 类别匹配加分
        
        # 价格范围匹配
        if item_features.get("price_range") == user_preferences.get("price_range"):  # 价格匹配
            score += 0.3  # 价格匹配加分
        
        # 标签匹配（简化）
        item_tags = set(item_features.get("tags", []))  # 商品标签
        user_tags = set(user_preferences.get("tags", []))  # 用户标签
        if item_tags and user_tags:
            common_tags = item_tags.intersection(user_tags)  # 共同标签
            if common_tags:
                score += 0.2 * len(common_tags)  # 每个共同标签加0.2分
        
        return min(score, 1.0)  # 返回分数，最大为1.0

class HybridFilteringCluster(BaseRecommendationCluster):
    """混合过滤集群"""
    
    def __init__(self):
        super().__init__("混合过滤集群")  # 调用父类初始化
        self.collaborative = CollaborativeFilteringCluster()  # 协同过滤
        self.content_based = ContentBasedFilteringCluster()  # 内容过滤
        self.context_aware = ContextAwareCluster()  # 上下文感知
    
    def generate_recommendations(self, user_id: str, user_profile: Dict[str, Any], 
                                interaction_history: List[Dict[str, Any]], 
                                context: Dict[str, Any], analysis: Dict[str, Any]) -> List[Dict[str, Any]]:
        """生成混合过滤推荐"""
        print(f"[{self.name}] 为用户 {user_id} 生成混合过滤推荐...")  # 开始日志
        
        # 从各个子集群获取推荐
        all_recommendations = []  # 所有推荐
        
        # 协同过滤推荐
        collaborative_recs = self.collaborative.generate_recommendations(  # 协同过滤推荐
            user_id, user_profile, interaction_history, context, analysis
        )
        all_recommendations.extend(collaborative_recs)  # 添加到总列表
        
        # 内容过滤推荐
        content_recs = self.content_based.generate_recommendations(  # 内容过滤推荐
            user_id, user_profile, interaction_history, context, analysis
        )
        all_recommendations.extend(content_recs)  # 添加到总列表
        
        # 上下文感知推荐
        context_recs = self.context_aware.generate_recommendations(  # 上下文感知推荐
            user_id, user_profile, interaction_history, context, analysis
        )
        all_recommendations.extend(context_recs)  # 添加到总列表
        
        # 混合策略：加权整合
        hybrid_recommendations = []
        for rec in all_recommendations:
            # 根据来源调整置信度
            source = rec.get("generated_by", "")  # 获取来源
            base_confidence = rec.get("confidence", 0.5)  # 基础置信度
            
            # 不同来源的权重
            weights = {
                "协同过滤集群": 0.4,    # 协同过滤权重
                "内容过滤集群": 0.3,    # 内容过滤权重
                "上下文感知集群": 0.3    # 上下文感知权重
            }
            
            weight = weights.get(source, 0.33)  # 获取权重
            adjusted_confidence = base_confidence * weight  # 调整置信度
            
            # 创建混合推荐
            hybrid_rec = rec.copy()  # 复制推荐
            hybrid_rec["confidence"] = adjusted_confidence  # 更新置信度
            hybrid_rec["hybrid_weight"] = weight  # 添加混合权重
            hybrid_rec["original_source"] = source  # 添加原始来源
            
            hybrid_recommendations.append(hybrid_rec)  # 添加到混合推荐列表
        
        # 按调整后的置信度排序
        hybrid_recommendations.sort(key=lambda x: x.get("confidence", 0), reverse=True)  # 按置信度降序
        
        print(f"[{self.name}] 生成 {len(hybrid_recommendations)} 个混合过滤推荐")  # 完成日志
        
        return hybrid_recommendations[:8]  # 返回前8个推荐

### 2.1 ACL消息结构

#### 核心概念
**"Agent间通信的标准消息格式"**

#### FIPA ACL消息结构
```
┌─────────────────────────────────────┐
│          消息头(Message Header)      │
│  • performative: 通信动作类型        │
│  • sender: 发送者Agent标识           │
│  • receiver: 接收者Agent标识         │
│  • reply-to: 回复地址                │
│  • content: 消息内容                 │
│  • language: 内容语言                │
│  • ontology: 使用的本体              │
│  • protocol: 通信协议                │
│  • conversation-id: 会话ID           │
│  • reply-with: 回复标识              │
│  • in-reply-to: 回复哪个消息         │
│  • encoding: 编码方式                │
└─────────────────────────────────────┘
```

# ACL消息实际实现
from typing import Dict, Any, Optional
from datetime import datetime
import json

class FIPAACLMessage:
    """FIPA ACL消息实现"""
    
    # 通信动作类型
    PERFORMATIVES = {
        "inform": "通知",          # 提供信息
        "request": "请求",         # 请求执行动作
        "agree": "同意",           # 同意请求
        "refuse": "拒绝",          # 拒绝请求
        "failure": "失败",         # 动作执行失败
        "not_understood": "不理解", # 不理解消息
        "cfp": "征求建议",         # 征求建议
        "propose": "提议",         # 提出建议
        "accept_proposal": "接受提议", # 接受提议
        "reject_proposal": "拒绝提议", # 拒绝提议
        "query_if": "查询是否",     # 查询事实
        "query_ref": "查询引用",    # 查询引用
        "subscribe": "订阅",       # 订阅信息
        "cancel": "取消"           # 取消订阅
    }
    
    def __init__(self, performative: str, sender: str, receiver: str, 
                 content: Any = None, conversation_id: Optional[str] = None):
        """初始化ACL消息"""
        if performative not in self.PERFORMATIVES:
            raise ValueError(f"无效的通信动作: {performative}")  # 无效动作错误
        
        self.performative = performative          # 通信动作
        self.sender = sender                      # 发送者
        self.receiver = receiver                  # 接收者
        self.content = content                    # 消息内容
        self.language = "FIPA-SL0"                # 语言：FIPA语义语言
        self.ontology = "default"                 # 本体
        self.encoding = "json"                    # 编码方式
        self.protocol = None                      # 协议
        self.reply_with = None                    # 回复标识
        self.in_reply_to = None                   # 回复哪个消息
        self.reply_by = None                      # 回复截止时间
        
        # 会话ID
        if conversation_id:
            self.conversation_id = conversation_id  # 使用提供的会话ID
        else:
            self.conversation_id = f"conv_{datetime.now().timestamp()}"  # 生成新会话ID
        
        # 消息ID
        self.message_id = f"msg_{datetime.now().timestamp()}_{hash(str(content))}"  # 消息ID
        
        # 时间戳
        self.timestamp = datetime.now().isoformat()  # 时间戳
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        return {
            "message_id": self.message_id,          # 消息ID
            "performative": self.performative,      # 通信动作
            "sender": self.sender,                  # 发送者
            "receiver": self.receiver,              # 接收者
            "content": self.content,                # 消息内容
            "language": self.language,              # 语言
            "ontology": self.ontology,              # 本体
            "encoding": self.encoding,              # 编码方式
            "conversation_id": self.conversation_id, # 会话ID
            "reply_with": self.reply_with,          # 回复标识
            "in_reply_to": self.in_reply_to,        # 回复哪个消息
            "reply_by": self.reply_by,              # 回复截止时间
            "timestamp": self.timestamp             # 时间戳
        }
    
    def to_json(self) -> str:
        """转换为JSON字符串"""
        return json.dumps(self.to_dict(), ensure_ascii=False, indent=2)  # 转换为JSON
    
    @classmethod
    def from_json(cls, json_str: str) -> 'FIPAACLMessage':
        """从JSON创建消息"""
        data = json.loads(json_str)  # 解析JSON
        
        # 创建消息实例
        message = cls(  # 创建消息
            performative=data["performative"],  # 通信动作
            sender=data["sender"],              # 发送者
            receiver=data["receiver"],          # 接收者
            content=data.get("content"),        # 消息内容
            conversation_id=data.get("conversation_id")  # 会话ID
        )
        
        # 设置其他属性
        message.language = data.get("language", message.language)  # 语言
        message.ontology = data.get("ontology", message.ontology)  # 本体
        message.encoding = data.get("encoding", message.encoding)  # 编码方式
        message.message_id = data.get("message_id", message.message_id)  # 消息ID
        message.reply_with = data.get("reply_with")  # 回复标识
        message.in_reply_to = data.get("in_reply_to")  # 回复哪个消息
        message.reply_by = data.get("reply_by")  # 回复截止时间
        message.timestamp = data.get("timestamp", message.timestamp)  # 时间戳
        
        return message  # 返回消息实例
    
    def create_reply(self, performative: str, content: Any = None) -> 'FIPAACLMessage':
        """创建回复消息"""
        reply = FIPAACLMessage(  # 创建回复消息
            performative=performative,  # 通信动作
            sender=self.receiver,       # 发送者变为原接收者
            receiver=self.sender,       # 接收者变为原发送者
            content=content,            # 回复内容
            conversation_id=self.conversation_id  # 保持相同会话ID
        )
        
        # 设置回复关系
        reply.in_reply_to = self.message_id  # 回复哪个消息
        reply.reply_with = f"reply_to_{self.message_id}"  # 回复标识
        
        return reply  # 返回回复消息
    
    def __str__(self) -> str:
        """字符串表示"""
        performative_cn = self.PERFORMATIVES.get(self.performative, self.performative)  # 中文动作
        return f"[{self.performative}({performative_cn})] {self.sender} → {self.receiver}: {str(self.content)[:50]}..."  # 字符串表示

# ACL消息管理器
class ACLMessageManager:
    """ACL消息管理器"""
    
    def __init__(self, agent_name: str):
        self.agent_name = agent_name  # Agent名称
        self.message_queue = []  # 消息队列
        self.conversations = {}  # 会话记录
        self.handlers = {}  # 消息处理器
    
    def register_handler(self, performative: str, handler):
        """注册消息处理器"""
        self.handlers[performative] = handler  # 注册处理器
        print(f"[{self.agent_name}] 注册 {performative} 消息处理器")  # 注册日志
    
    def send_message(self, message: FIPAACLMessage):
        """发送消息（模拟）"""
        print(f"[{self.agent_name}] 发送消息: {message}")  # 发送日志
        
        # 在实际系统中，这里会将消息发送到消息总线或其他Agent
        # 模拟：直接调用接收者的receive_message方法
        return True  # 返回发送成功
    
    def receive_message(self, message: FIPAACLMessage):
        """接收消息"""
        print(f"[{self.agent_name}] 收到消息: {message}")  # 接收日志
        
        # 添加到消息队列
        self.message_queue.append(message)  # 添加到队列
        
        # 记录会话
        conv_id = message.conversation_id  # 会话ID
        if conv_id not in self.conversations:
            self.conversations[conv_id] = []  # 初始化会话记录
        
        self.conversations[conv_id].append({  # 添加消息记录
            "direction": "incoming",  # 方向：入站
            "message": message.to_dict(),  # 消息内容
            "timestamp": datetime.now().isoformat()  # 时间戳
        })
        
        # 处理消息
        self._process_message(message)  # 处理消息
    
    def _process_message(self, message: FIPAACLMessage):
        """处理消息"""
        performative = message.performative  # 通信动作
        
        if performative in self.handlers:
            # 调用注册的处理器
            handler = self.handlers[performative]  # 获取处理器
            response = handler(message)  # 处理消息
            
            if response:
                # 发送回复
                self.send_message(response)  # 发送回复
        else:
            # 没有处理器，发送"不理解"回复
            reply = message.create_reply("not_understood", "没有对应的消息处理器")  # 创建不理解回复
            self.send_message(reply)  # 发送回复
    
    def process_queue(self):
        """处理消息队列"""
        while self.message_queue:
            message = self.message_queue.pop(0)  # 取出消息
            self._process_message(message)  # 处理消息
    
    def get_conversation_history(self, conversation_id: str) -> List[Dict[str, Any]]:
        """获取会话历史"""
        return self.conversations.get(conversation_id, [])  # 返回会话历史

# 基于市场的协调实际实现
from typing import Dict, List, Any, Tuple
import random
from datetime import datetime

class TaskMarket:
    """任务市场：基于拍卖的协调机制"""
    
    def __init__(self):
        self.tasks = []  # 任务列表
        self.bids = {}   # 投标记录 {task_id: [(agent, bid_amount, bid_time)]}
        self.assignments = {}  # 任务分配 {task_id: (agent, bid_amount)}
        self.auction_history = []  # 拍卖历史
    
    def list_task(self, task_id: str, task_description: str, reserve_price: float = 0.0):
        """列出任务"""
        task = {
            "task_id": task_id,            # 任务ID
            "description": task_description, # 任务描述
            "reserve_price": reserve_price, # 保留价
            "status": "open",              # 状态：开放
            "listed_at": datetime.now().isoformat()  # 列出时间
        }
        
        self.tasks.append(task)  # 添加到任务列表
        self.bids[task_id] = []  # 初始化投标记录
        
        print(f"[市场] 任务已列出: {task_id} - {task_description} (保留价: {reserve_price})")  # 列出日志
        
        return task_id  # 返回任务ID
    
    def submit_bid(self, agent: 'MarketAgent', task_id: str, bid_amount: float) -> bool:
        """提交投标"""
        if task_id not in self.bids:
            print(f"[市场] 未知任务: {task_id}")  # 未知任务警告
            return False
        
        # 检查任务状态
        task = self._get_task(task_id)  # 获取任务
        if task["status"] != "open":  # 如果任务不开放
            print(f"[市场] 任务 {task_id} 已关闭")  # 任务关闭警告
            return False
        
        # 检查投标是否高于保留价
        if bid_amount < task["reserve_price"]:  # 如果投标低于保留价
            print(f"[市场] 投标 {bid_amount} 低于保留价 {task['reserve_price']}")  # 投标过低警告
            return False
        
        # 记录投标
        bid_record = {
            "agent": agent,                # Agent
            "bid_amount": bid_amount,      # 投标金额
            "submitted_at": datetime.now().isoformat()  # 提交时间
        }
        
        self.bids[task_id].append(bid_record)  # 添加到投标记录
        
        print(f"[市场] {agent.name} 对任务 {task_id} 投标: {bid_amount}")  # 投标日志
        
        return True  # 返回成功
    
    def close_auction(self, task_id: str) -> Tuple[bool, Any]:
        """关闭拍卖，选择中标者"""
        if task_id not in self.bids:
            print(f"[市场] 未知任务: {task_id}")  # 未知任务警告
            return False, None
        
        bids = self.bids[task_id]  # 获取投标记录
        if not bids:  # 如果没有投标
            print(f"[市场] 任务 {task_id} 无投标")  # 无投标日志
            return False, None
        
        # 选择最高投标
        winning_bid = max(bids, key=lambda x: x["bid_amount"])  # 选择最高投标
        
        # 更新任务状态
        task = self._get_task(task_id)  # 获取任务
        task["status"] = "assigned"  # 状态：已分配
        task["assigned_to"] = winning_bid["agent"].name  # 分配给
        task["winning_bid"] = winning_bid["bid_amount"]  # 中标金额
        task["closed_at"] = datetime.now().isoformat()  # 关闭时间
        
        # 记录分配
        self.assignments[task_id] = {  # 记录分配
            "agent": winning_bid["agent"],      # Agent
            "bid_amount": winning_bid["bid_amount"],  # 投标金额
            "assigned_at": datetime.now().isoformat()  # 分配时间
        }
        
        # 记录拍卖历史
        auction_record = {
            "task_id": task_id,  # 任务ID
            "total_bids": len(bids),  # 总投标数
            "winning_agent": winning_bid["agent"].name,  # 中标Agent
            "winning_bid": winning_bid["bid_amount"],  # 中标金额
            "closed_at": datetime.now().isoformat()  # 关闭时间
        }
        self.auction_history.append(auction_record)  # 添加到历史
        
        print(f"[市场] 任务 {task_id} 分配给 {winning_bid['agent'].name} (投标: {winning_bid['bid_amount']})")  # 分配日志
        
        return True, winning_bid  # 返回成功和中标信息
    
    def _get_task(self, task_id: str) -> Dict[str, Any]:
        """获取任务"""
        for task in self.tasks:
            if task["task_id"] == task_id:  # 如果任务ID匹配
                return task  # 返回任务
        return None  # 返回None
    
    def get_market_status(self) -> Dict[str, Any]:
        """获取市场状态"""
        open_tasks = [t for t in self.tasks if t["status"] == "open"]  # 开放任务
        assigned_tasks = [t for t in self.tasks if t["status"] == "assigned"]  # 已分配任务
        
        return {
            "total_tasks": len(self.tasks),  # 总任务数
            "open_tasks": len(open_tasks),   # 开放任务数
            "assigned_tasks": len(assigned_tasks),  # 已分配任务数
            "total_assignments": len(self.assignments),  # 总分配数
            "recent_auctions": self.auction_history[-5:] if self.auction_history else []  # 最近拍卖
        }

class MarketAgent:
    """市场Agent：参与拍卖的Agent"""
    
    def __init__(self, name: str, skill_set: List[str], budget: float = 100.0):
        self.name = name  # Agent名称
        self.skill_set = skill_set  # 技能集合
        self.budget = budget  # 预算
        self.assigned_tasks = []  # 分配的任务
        self.earnings = 0.0  # 收益
        self.bid_history = []  # 投标历史
    
    def evaluate_task(self, task_description: str) -> Tuple[bool, float]:
        """评估任务，返回是否适合和投标金额"""
        # 简单评估：基于技能匹配度
        skill_match = 0  # 技能匹配度
        
        for skill in self.skill_set:  # 遍历技能
            if skill in task_description.lower():  # 如果技能在任务描述中
                skill_match += 1  # 增加匹配度
        
        if skill_match == 0:  # 如果没有技能匹配
            return False, 0.0  # 返回不适合
        
        # 计算投标金额（基于匹配度和预算）
        base_bid = 10.0 * skill_match  # 基础投标
        max_bid = min(base_bid * 2, self.budget * 0.5)  # 最大投标（不超过预算的50%）
        
        # 随机投标金额（在实际中会有更复杂的策略）
        bid_amount = random.uniform(base_bid, max_bid)  # 随机投标
        
        return True, bid_amount  # 返回适合和投标金额
    
    def submit_bid_to_market(self, market: TaskMarket, task_id: str, task_description: str) -> bool:
        """向市场提交投标"""
        # 评估任务
        suitable, bid_amount = self.evaluate_task(task_description)  # 评估任务
        
        if not suitable:  # 如果不适合
            print(f"[{self.name}] 任务 {task_id} 不适合我的技能")  # 不适合日志
            return False  # 返回失败
        
        # 提交投标
        success = market.submit_bid(self, task_id, bid_amount)  # 提交投标
        
        if success:  # 如果成功
            # 记录投标历史
            bid_record = {
                "task_id": task_id,            # 任务ID
                "bid_amount": bid_amount,      # 投标金额
                "task_description": task_description,  # 任务描述
                "submitted_at": datetime.now().isoformat()  # 提交时间
            }
            self.bid_history.append(bid_record)  # 添加到投标历史
            print(f"[{self.name}] 成功提交投标: {bid_amount}")  # 成功日志
        
        return success  # 返回成功状态
    
    def receive_task_assignment(self, task_id: str, task_description: str, bid_amount: float):
        """接收任务分配"""
        assignment = {
            "task_id": task_id,            # 任务ID
            "description": task_description, # 任务描述
            "bid_amount": bid_amount,      # 投标金额
            "assigned_at": datetime.now().isoformat()  # 分配时间
        }
        
        self.assigned_tasks.append(assignment)  # 添加到分配任务
        self.earnings += bid_amount  # 增加收益
        self.budget -= bid_amount * 0.1  # 扣除成本（假设成本的10%）
        
        print(f"[{self.name}] 收到任务分配: {task_id} (收益: {bid_amount})")  # 分配日志
    
    def get_agent_status(self) -> Dict[str, Any]:
        """获取Agent状态"""
        return {
            "name": self.name,  # 名称
            "skills": self.skill_set,  # 技能
            "budget": self.budget,  # 预算
            "earnings": self.earnings,  # 收益
            "assigned_tasks": len(self.assigned_tasks),  # 分配任务数
            "total_bids": len(self.bid_history),  # 总投标数
            "success_rate": len(self.assigned_tasks) / max(1, len(self.bid_history))  # 成功率
        }

class ConventionBasedCoordination:

#### 实际代码实现
```python
# 基于约定的协调实际实现
from typing import Dict, List, Any, Optional
from enum import Enum

class ConventionRuleType(Enum):
    """约定规则类型"""
    PRIORITY_BASED = "priority_based"      # 基于优先级
    ROUND_ROBIN = "round_robin"            # 轮询
    LOAD_BALANCING = "load_balancing"      # 负载均衡
    TIME_SLOT = "time_slot"                # 时间片
    RESOURCE_SHARING = "resource_sharing"  # 资源共享

class ConventionRule:
    """约定规则"""
    
    def __init__(self, rule_type: ConventionRuleType, condition: Dict[str, Any], action: Dict[str, Any]):
        self.rule_type = rule_type  # 规则类型
        self.condition = condition  # 应用条件
        self.action = action        # 执行动作
        self.usage_count = 0        # 使用次数
    
    def applies_to(self, context: Dict[str, Any]) -> bool:
        """检查规则是否适用于当前上下文"""
        self.usage_count += 1  # 增加使用计数
        
        # 检查条件
        for key, expected_value in self.condition.items():
            actual_value = context.get(key)  # 获取实际值
            
            if actual_value != expected_value:  # 如果不匹配
                return False  # 返回不适用
        
        return True  # 返回适用
    
    def apply(self, agents: List['ConventionAgent'], resources: List[str]) -> Dict[str, Any]:
        """应用规则"""
        if self.rule_type == ConventionRuleType.PRIORITY_BASED:  # 基于优先级
            return self._apply_priority_based(agents, resources)  # 应用基于优先级
        elif self.rule_type == ConventionRuleType.ROUND_ROBIN:  # 轮询
            return self._apply_round_robin(agents, resources)  # 应用轮询
        elif self.rule_type == ConventionRuleType.LOAD_BALANCING:  # 负载均衡
            return self._apply_load_balancing(agents, resources)  # 应用负载均衡
        elif self.rule_type == ConventionRuleType.TIME_SLOT:  # 时间片
            return self._apply_time_slot(agents, resources)  # 应用时间片
        elif self.rule_type == ConventionRuleType.RESOURCE_SHARING:  # 资源共享
            return self._apply_resource_sharing(agents, resources)  # 应用资源共享
        else:
            return {"error": f"未知规则类型: {self.rule_type}"}  # 未知规则类型
    
    def _apply_priority_based(self, agents: List['ConventionAgent'], resources: List[str]) -> Dict[str, Any]:
        """基于优先级的分配"""
        # 按优先级排序
        sorted_agents = sorted(agents, key=lambda a: a.priority, reverse=True)  # 按优先级降序
        
        allocation = {}  # 分配结果
        
        for i, agent in enumerate(sorted_agents):  # 遍历排序后的Agent
            if i < len(resources):  # 如果还有资源
                allocation[agent.name] = resources[i]  # 分配资源
            else:
                allocation[agent.name] = None  # 没有资源
        
        return {
            "rule_type": "priority_based",  # 规则类型
            "allocation": allocation,       # 分配结果
            "sorted_agents": [a.name for a in sorted_agents]  # 排序后的Agent
        }
    
    def _apply_round_robin(self, agents: List['ConventionAgent'], resources: List[str]) -> Dict[str, Any]:
        """轮询分配"""
        allocation = {}  # 分配结果
        
        # 简单轮询
        for i, agent in enumerate(agents):  # 遍历Agent
            resource_index = i % len(resources)  # 计算资源索引
            allocation[agent.name] = resources[resource_index]  # 分配资源
        
        return {
            "rule_type": "round_robin",  # 规则类型
            "allocation": allocation,    # 分配结果
            "distribution": "circular"   # 分布方式：循环
        }
    
    def _apply_load_balancing(self, agents: List['ConventionAgent'], resources: List[str]) -> Dict[str, Any]:
        """负载均衡分配"""
        # 按当前负载排序（负载低的优先）
        sorted_agents = sorted(agents, key=lambda a: a.current_load)  # 按负载升序
        
        allocation = {}  # 分配结果
        
        # 平均分配资源
        resources_per_agent = len(resources) // len(agents)  # 每个Agent的资源数
        remainder = len(resources) % len(agents)  # 剩余资源
        
        resource_index = 0  # 资源索引
        for i, agent in enumerate(sorted_agents):  # 遍历排序后的Agent
            # 计算分配给该Agent的资源数
            count = resources_per_agent  # 基础数量
            if i < remainder:  # 如果还有剩余资源
                count += 1  # 增加一个
            
            # 分配资源
            agent_resources = resources[resource_index:resource_index + count]  # Agent的资源
            allocation[agent.name] = agent_resources  # 分配
            
            resource_index += count  # 更新资源索引
        
        return {
            "rule_type": "load_balancing",  # 规则类型
            "allocation": allocation,       # 分配结果
            "resources_per_agent": resources_per_agent,  # 每个Agent的资源数
            "remainder_distributed": remainder  # 分配的剩余资源数
        }
    
    def _apply_time_slot(self, agents: List['ConventionAgent'], resources: List[str]) -> Dict[str, Any]:
        """时间片分配"""
        # 简单时间片：每个Agent分配一个时间片
        from datetime import datetime
        
        time_slots = []  # 时间片列表
        current_time = datetime.now()  # 当前时间
        
        # 生成时间片（每个资源分配一个时间片）
        for i, resource in enumerate(resources):  # 遍历资源
            slot_start = current_time.replace(hour=9 + i, minute=0, second=0)  # 开始时间
            slot_end = slot_start.replace(hour=10 + i, minute=0, second=0)  # 结束时间
            
            time_slots.append({  # 添加时间片
                "resource": resource,  # 资源
                "start": slot_start.isoformat(),  # 开始时间
                "end": slot_end.isoformat(),  # 结束时间
                "duration_hours": 1  # 持续时间
            })
        
        # 分配时间片给Agent
        allocation = {}  # 分配结果
        for i, agent in enumerate(agents):  # 遍历Agent
            if i < len(time_slots):  # 如果还有时间片
                allocation[agent.name] = time_slots[i]  # 分配时间片
            else:
                allocation[agent.name] = None  # 没有时间片
        
        return {
            "rule_type": "time_slot",  # 规则类型
            "allocation": allocation,  # 分配结果
            "time_slots": time_slots   # 时间片列表
        }
    
    def _apply_resource_sharing(self, agents: List['ConventionAgent'], resources: List[str]) -> Dict[str, Any]:
        """资源共享分配"""
        # 所有Agent共享所有资源
        allocation = {}  # 分配结果
        
        for agent in agents:  # 遍历Agent
            allocation[agent.name] = resources.copy()  # 分配所有资源
        
        return {
            "rule_type": "resource_sharing",  # 规则类型
            "allocation": allocation,         # 分配结果
            "sharing_mode": "full_shared"     # 共享模式：完全共享
        }
    
    def get_rule_info(self) -> Dict[str, Any]:
        """获取规则信息"""
        return {
            "rule_type": self.rule_type.value,  # 规则类型
            "condition": self.condition,        # 条件
            "action": self.action,              # 动作
            "usage_count": self.usage_count     # 使用次数
        }

class ConventionAgent:
    """约定Agent：遵循约定规则的Agent"""
    
    def __init__(self, name: str, priority: int = 1, current_load: int = 0):
        self.name = name  # Agent名称
        self.priority = priority  # 优先级（1-10，越高越优先）
        self.current_load = current_load  # 当前负载
        self.assigned_resources = []  # 分配的资源
        self.convention_history = []  # 约定历史
    
    def request_coordination(self, convention_system: 'ConventionSystem', 
                           context: Dict[str, Any], resources: List[str]) -> Dict[str, Any]:
        """请求协调"""
        print(f"[{self.name}] 请求协调，上下文: {context}")  # 请求日志
        
        # 调用约定系统进行协调
        result = convention_system.coordinate([self], context, resources)  # 协调
        
        if "allocation" in result and self.name in result["allocation"]:  # 如果有分配结果
            assigned_resource = result["allocation"][self.name]  # 分配的资源
            
            if assigned_resource:  # 如果有分配资源
                if isinstance(assigned_resource, list):  # 如果是列表
                    self.assigned_resources.extend(assigned_resource)  # 添加到分配资源
                else:
                    self.assigned_resources.append(assigned_resource)  # 添加到分配资源
                
                # 更新负载
                self.current_load += 1  # 增加负载
            
            # 记录约定历史
            history_record = {
                "timestamp": datetime.now().isoformat(),  # 时间戳
                "context": context,                       # 上下文
                "resources_requested": resources,         # 请求的资源
                "allocation_result": result,              # 分配结果
                "assigned_resources": assigned_resource   # 分配的资源
            }
            self.convention_history.append(history_record)  # 添加到历史
        
        return result  # 返回结果
    
    def get_agent_status(self) -> Dict[str, Any]:
        """获取Agent状态"""
        return {
            "name": self.name,  # 名称
            "priority": self.priority,  # 优先级
            "current_load": self.current_load,  # 当前负载
            "assigned_resources": self.assigned_resources,  # 分配的资源
            "convention_requests": len(self.convention_history)  # 约定请求数
        }

class ConventionSystem:
    """约定系统：管理约定规则"""
    
    def __init__(self):
        self.rules = []  # 规则列表
        self.coordination_history = []  # 协调历史
        
        # 初始化默认规则
        self._initialize_default_rules()  # 初始化默认规则
    
    def _initialize_default_rules(self):
        """初始化默认规则"""
        # 规则1：高优先级任务使用基于优先级的规则
        self.rules.append(ConventionRule(  # 添加规则
            rule_type=ConventionRuleType.PRIORITY_BASED,  # 规则类型：基于优先级
            condition={"task_type": "high_priority", "resource_type": "exclusive"},  # 条件：高优先级任务，独占资源
            action={"allocation_method": "priority_based"}  # 动作：基于优先级分配
        ))
        
        # 规则2：常规任务使用轮询规则
        self.rules.append(ConventionRule(  # 添加规则
            rule_type=ConventionRuleType.ROUND_ROBIN,  # 规则类型：轮询
            condition={"task_type": "regular", "resource_type": "shared"},  # 条件：常规任务，共享资源
            action={"allocation_method": "round_robin"}  # 动作：轮询分配
        ))
        
        # 规则3：计算密集型任务使用负载均衡
        self.rules.append(ConventionRule(  # 添加规则
            rule_type=ConventionRuleType.LOAD_BALANCING,  # 规则类型：负载均衡
            condition={"task_type": "compute_intensive", "load_aware": True},  # 条件：计算密集型，负载感知
            action={"allocation_method": "load_balancing"}  # 动作：负载均衡分配
        ))
        
        # 规则4：时间敏感任务使用时间片
        self.rules.append(ConventionRule(  # 添加规则
            rule_type=ConventionRuleType.TIME_SLOT,  # 规则类型：时间片
            condition={"task_type": "time_sensitive", "requires_scheduling": True},  # 条件：时间敏感，需要调度
            action={"allocation_method": "time_slot"}  # 动作：时间片分配
        ))
        
        # 规则5：协作任务使用资源共享
        self.rules.append(ConventionRule(  # 添加规则
            rule_type=ConventionRuleType.RESOURCE_SHARING,  # 规则类型：资源共享
            condition={"task_type": "collaborative", "resource_type": "shared"},  # 条件：协作任务，共享资源
            action={"allocation_method": "resource_sharing"}  # 动作：资源共享
        ))
        
        print("[约定系统] 默认规则已初始化")  # 初始化日志
    
    def add_rule(self, rule: ConventionRule):
        """添加规则"""
        self.rules.append(rule)  # 添加规则
        print(f"[约定系统] 添加规则: {rule.rule_type.value}")  # 添加日志
    
    def coordinate(self, agents: List[ConventionAgent], context: Dict[str, Any], 
                  resources: List[str]) -> Dict[str, Any]:
        """协调Agent和资源"""
        print(f"[约定系统] 开始协调 {len(agents)} 个Agent, {len(resources)} 个资源")  # 开始协调日志
        
        # 查找适用的规则
        applicable_rules = []  # 适用规则列表
        
        for rule in self.rules:  # 遍历规则
            if rule.applies_to(context):  # 如果规则适用
                applicable_rules.append(rule)  # 添加到适用规则列表
        
        if not applicable_rules:  # 如果没有适用规则
            print("[约定系统] 没有适用规则，使用默认轮询")  # 无规则日志
            # 使用默认规则：轮询
            default_rule = ConventionRule(  # 创建默认规则
                rule_type=ConventionRuleType.ROUND_ROBIN,  # 规则类型：轮询
                condition={},  # 空条件
                action={"allocation_method": "default_round_robin"}  # 动作：默认轮询
            )
            applicable_rules.append(default_rule)  # 添加到适用规则列表
        
        # 应用第一个适用规则（在实际中可能会有更复杂的规则选择逻辑）
        selected_rule = applicable_rules[0]  # 选择第一个适用规则
        print(f"[约定系统] 选择规则: {selected_rule.rule_type.value}")  # 规则选择日志
        
        # 应用规则
        result = selected_rule.apply(agents, resources)  # 应用规则
        
        # 记录协调历史
        history_record = {
            "timestamp": datetime.now().isoformat(),  # 时间戳
            "agents": [a.name for a in agents],       # Agent列表
            "resources": resources,                   # 资源列表
            "context": context,                       # 上下文
            "selected_rule": selected_rule.get_rule_info(),  # 选择的规则
            "result": result                          # 结果
        }
        self.coordination_history.append(history_record)  # 添加到历史
        
        print(f"[约定系统] 协调完成，使用规则: {selected_rule.rule_type.value}")  # 完成日志
        
        return result  # 返回结果
    
    def get_system_status(self) -> Dict[str, Any]:
        """获取系统状态"""
        rule_usage = {}  # 规则使用情况
        
        for rule in self.rules:  # 遍历规则
            rule_usage[rule.rule_type.value] = rule.usage_count  # 规则使用次数
        
        return {
            "total_rules": len(self.rules),  # 总规则数
            "rule_usage": rule_usage,        # 规则使用情况
            "total_coordinations": len(self.coordination_history),  # 总协调次数
            "recent_coordinations": self.coordination_history[-3:] if self.coordination_history else []  # 最近协调
        }
