## 一、分层架构(Hierarchical Architecture) - 10分钟

### 架构图示
```
┌─────────────────────────────────────┐
│          协调层(Coordinator)         │  ← 顶层控制（管理整体流程）
├─────────────────────────────────────┤
│          规划层(Planner)             │  ← 任务分解（制定执行计划）
├─────────────────────────────────────┤
│          执行层(Executor)            │  ← 工具调用（执行具体操作）
├─────────────────────────────────────┤
│          感知层(Perceptor)           │  ← 环境感知（获取输入信息）
└─────────────────────────────────────┘
```

### 各层核心功能（记忆要点）
1. **协调层** (顶层) - 像项目经理
   - 任务分配与优先级管理
   - 资源冲突解决
   - 整体进度监控

2. **规划层** (策略层) - 像架构师
   - 复杂任务分解为子任务
   - 执行路径规划
   - 备选策略制定

3. **执行层** (操作层) - 像开发工程师
   - 具体工具调用
   - 动作执行与状态监控
   - 异常处理

4. **感知层** (输入层) - 像产品经理
   - 环境信息采集
   - 数据预处理
   - 状态识别

### 1.1 伪代码示例（理解架构思想）
```python
# 分层架构伪代码示例 - 理解架构思想
class HierarchicalAgent:
    def process(self, input_data):
        # 1. 感知层：获取并处理输入数据
        perceived = self.perceptor.observe(input_data)  # 观察环境，获取输入
        
        # 2. 规划层：基于感知结果制定详细计划
        plan = self.planner.create_plan(perceived)      # 创建执行计划
        
        # 3. 执行层：按照计划执行具体操作
        result = self.executor.execute(plan)            # 执行计划
        
        # 4. 协调层：监控执行结果并调整策略
        self.coordinator.monitor(result)                # 监控和调整
        return result                                   # 返回最终结果
```

### 1.2 LangChain代码案例（实际实现）
```python
# 分层架构的LangChain实现 - 文档问答系统示例
from langchain.agents import AgentExecutor, create_react_agent
from langchain.tools import Tool
from langchain_community.llms import OpenAI
from langchain.chains import LLMChain
from langchain.prompts import PromptTemplate
import os

# 设置API密钥（实际使用时需要配置）
os.environ["OPENAI_API_KEY"] = "your-api-key-here"

# ========== 1. 感知层：文档加载和预处理 ==========
class DocumentPerceptor:
    """感知层：负责文档加载和预处理"""
    def observe(self, document_path):
        # 加载文档
        from langchain_community.document_loaders import TextLoader
        loader = TextLoader(document_path)      # 创建文档加载器
        documents = loader.load()               # 加载文档内容
        
        # 文档分割（便于后续处理）
        from langchain.text_splitter import CharacterTextSplitter
        text_splitter = CharacterTextSplitter(chunk_size=1000, chunk_overlap=0)
        docs = text_splitter.split_documents(documents)  # 分割文档
        
        return docs  # 返回处理后的文档

# ========== 2. 规划层：问题分析和计划制定 ==========
class QueryPlanner:
    """规划层：分析问题并制定回答计划"""
    def create_plan(self, query, docs):
        # 创建LLM实例用于规划
        llm = OpenAI(temperature=0)  # 创建语言模型实例，temperature=0保证确定性
        
        # 规划提示模板
        plan_prompt = PromptTemplate(
            input_variables=["query", "doc_count"],
            template="""
            问题：{query}
            可用文档片段数量：{doc_count}
            
            请制定回答计划：
            1. 需要从文档中提取哪些关键信息？
            2. 如何组织回答结构？
            3. 是否需要补充外部知识？
            """
        )
        
        # 创建规划链
        plan_chain = LLMChain(llm=llm, prompt=plan_prompt)  # 创建语言模型链
        plan = plan_chain.run(query=query, doc_count=len(docs))  # 执行规划
        
        return plan  # 返回执行计划

# ========== 3. 执行层：具体工具调用和执行 ==========
class AnswerExecutor:
    """执行层：执行具体的文档检索和回答生成"""
    def execute(self, plan, query, docs):
        # 创建向量存储用于文档检索
        from langchain_community.embeddings import OpenAIEmbeddings
        from langchain_community.vectorstores import Chroma
        
        embeddings = OpenAIEmbeddings()  # 创建嵌入模型
        vectorstore = Chroma.from_documents(docs, embeddings)  # 创建向量存储
        
        # 创建检索链
        from langchain.chains import RetrievalQA
        qa_chain = RetrievalQA.from_chain_type(
            llm=OpenAI(),                     # 语言模型
            chain_type="stuff",               # 链类型：stuff（简单拼接）
            retriever=vectorstore.as_retriever()  # 检索器
        )
        
        # 执行回答生成
        answer = qa_chain.run(query)  # 运行问答链
        
        return answer  # 返回生成的答案

# ========== 4. 协调层：整体流程协调 ==========
class SystemCoordinator:
    """协调层：管理整个分层架构的流程"""
    def __init__(self):
        # 初始化各层组件
        self.perceptor = DocumentPerceptor()   # 感知层
        self.planner = QueryPlanner()          # 规划层
        self.executor = AnswerExecutor()       # 执行层
    
    def process_query(self, document_path, query):
        """处理查询的完整流程"""
        print("=== 协调层：开始处理查询 ===")
        
        # 1. 感知层：加载文档
        print("1. 感知层：加载和预处理文档...")
        docs = self.perceptor.observe(document_path)
        print(f"   加载了 {len(docs)} 个文档片段")
        
        # 2. 规划层：制定回答计划
        print("2. 规划层：分析问题并制定计划...")
        plan = self.planner.create_plan(query, docs)
        print(f"   生成的计划：\n{plan[:200]}...")  # 只显示前200字符
        
        # 3. 执行层：执行计划生成答案
        print("3. 执行层：执行计划生成答案...")
        answer = self.executor.execute(plan, query, docs)
        
        # 4. 协调层：监控和返回结果
        print("4. 协调层：整合结果...")
        result = {
            "query": query,
            "plan": plan,
            "answer": answer,
            "documents_processed": len(docs)
        }
        
        print("=== 协调层：处理完成 ===")
        return result

# ========== 使用示例 ==========
if __name__ == "__main__":
    # 创建分层架构系统
    hierarchical_system = SystemCoordinator()  # 创建协调层实例
    
    # 处理查询
    document_path = "example.txt"  # 文档路径
    user_query = "LangChain的主要功能是什么？"  # 用户查询
    
    # 执行完整流程
    result = hierarchical_system.process_query(document_path, user_query)  # 处理查询
    
    # 显示结果
    print("\n最终答案：")
    print(result["answer"])  # 打印答案
```

## 二、黑板架构(Blackboard Architecture) - 10分钟

### 核心概念（一句话理解）
**"多个专家共同解决一个问题，通过共享黑板交流"**

### 架构组成
```
    ┌─────────────────┐
    │   知识源A       │  ← 专家1（如：语义理解专家）
    │  (专家1)        │
    └────────┬────────┘
             ↓ 读取/写入
    ┌─────────────────┐
    │    黑板         │ ← 共享数据空间（存储中间结果）
    │  (Blackboard)   │
    └────────┬────────┘
             ↓ 读取/写入
    ┌─────────────────┐
    │   知识源B       │  ← 专家2（如：代码生成专家）
    │  (专家2)        │
    └─────────────────┘
```

```python
# 黑板架构的LangChain实现 - 多专家协作系统示例
from langchain.agents import Tool, AgentExecutor
from langchain.memory import ConversationBufferMemory
from langchain import OpenAI
import os

# 设置API密钥
os.environ["OPENAI_API_KEY"] = "your-api-key-here"

# ========== 1. 黑板类（共享数据空间） ==========
class Blackboard:
    """黑板：共享数据空间，存储中间结果"""
    def __init__(self):
        self.data = {}  # 存储所有共享数据
        self.history = []  # 记录所有操作历史
    
    def write(self, key, value, source="system"):
        """向黑板写入数据"""
        self.data[key] = value  # 存储数据
        self.history.append({  # 记录操作历史
            "action": "write",
            "key": key,
            "value": value[:100] if isinstance(value, str) else str(value)[:100],  # 截断显示
            "source": source
        })
        print(f"[黑板] {source} 写入: {key} = {value[:50]}...")  # 日志输出
    
    def read(self, key):
        """从黑板读取数据"""
        value = self.data.get(key, None)  # 获取数据
        if value is not None:
            print(f"[黑板] 读取: {key} = {str(value)[:50]}...")  # 日志输出
        return value
    
    def get_state(self):
        """获取当前黑板状态"""
        return {
            "keys": list(self.data.keys()),  # 所有键
            "history_count": len(self.history),  # 历史记录数量
            "last_action": self.history[-1] if self.history else None  # 最后操作
        }

# ========== 2. 知识源类（不同领域的专家） ==========
class LanguageUnderstandingExpert:
    """知识源1：语言理解专家"""
    def __init__(self, llm):
        self.llm = llm  # 语言模型
        self.name = "语言理解专家"  # 专家名称
    
    def contribute(self, blackboard, query):
        """贡献：分析查询的语义和意图"""
        print(f"[{self.name}] 开始分析查询语义...")
        
        # 从黑板读取查询
        current_query = blackboard.read("current_query") or query  # 读取查询
        
        # 语义分析提示
        analysis_prompt = f"""
        请分析以下查询的语义和意图：
        查询：{current_query}
        
        请分析：
        1. 查询的核心意图是什么？
        2. 涉及哪些关键概念？
        3. 需要什么类型的信息？
        """
        
        # 使用LLM进行分析
        analysis = self.llm(analysis_prompt)  # 调用语言模型
        
        # 将分析结果写入黑板
        blackboard.write("semantic_analysis", analysis, self.name)  # 写入语义分析
        blackboard.write("expert_contributions", f"{self.name}: 完成语义分析", self.name)  # 记录贡献
        
        return {"status": "completed", "expert": self.name}  # 返回状态

class CodeGenerationExpert:
    """知识源2：代码生成专家"""
    def __init__(self, llm):
        self.llm = llm  # 语言模型
        self.name = "代码生成专家"  # 专家名称
    
    def contribute(self, blackboard):
        """贡献：根据需求生成代码"""
        print(f"[{self.name}] 开始生成代码...")
        
        # 从黑板读取需求
        requirements = blackboard.read("requirements")  # 读取需求
        semantic_analysis = blackboard.read("semantic_analysis")  # 读取语义分析
        
        if not requirements:
            return {"status": "waiting", "reason": "等待需求明确"}  # 等待需求
        
        # 代码生成提示
        code_prompt = f"""
        根据以下需求生成Python代码：
        
        需求：{requirements}
        语义分析：{semantic_analysis}
        
        请生成：
        1. 完整的代码实现
        2. 必要的注释说明
        3. 使用示例
        """
        
        # 使用LLM生成代码
        generated_code = self.llm(code_prompt)  # 调用语言模型
        
        # 将生成的代码写入黑板
        blackboard.write("generated_code", generated_code, self.name)  # 写入生成代码
        blackboard.write("expert_contributions", f"{self.name}: 生成代码完成", self.name)  # 记录贡献
        
        return {"status": "completed", "expert": self.name}  # 返回状态

class ValidationExpert:
    """知识源3：验证专家"""
    def __init__(self, llm):
        self.llm = llm  # 语言模型
        self.name = "验证专家"  # 专家名称
    
    def contribute(self, blackboard):
        """贡献：验证解决方案的完整性"""
        print(f"[{self.name}] 开始验证解决方案...")
        
        # 从黑板读取所有相关信息
        generated_code = blackboard.read("generated_code")  # 读取生成代码
        requirements = blackboard.read("requirements")  # 读取需求
        
        if not generated_code:
            return {"status": "waiting", "reason": "等待代码生成"}  # 等待代码
        
        # 验证提示
        validation_prompt = f"""
        验证以下代码是否满足需求：
        
        需求：{requirements}
        生成的代码：{generated_code}
        
        请验证：
        1. 代码是否完整可运行？
        2. 是否满足所有需求？
        3. 需要哪些改进建议？
        """
        
        # 使用LLM进行验证
        validation_result = self.llm(validation_prompt)  # 调用语言模型
        
        # 将验证结果写入黑板
        blackboard.write("validation_result", validation_result, self.name)  # 写入验证结果
        blackboard.write("expert_contributions", f"{self.name}: 验证完成", self.name)  # 记录贡献
        blackboard.write("solution_status", "validated", self.name)  # 更新解决方案状态
        
        return {"status": "completed", "expert": self.name}  # 返回状态

# ========== 3. 控制器类（协调专家协作） ==========
class BlackboardController:
    """控制器：协调知识源协作"""
    def __init__(self):
        # 初始化黑板
        self.blackboard = Blackboard()  # 创建黑板
        
        # 初始化语言模型
        self.llm = OpenAI(temperature=0.7)  # 创建语言模型实例
        
        # 初始化知识源（专家）
        self.experts = [
            LanguageUnderstandingExpert(self.llm),  # 语言理解专家
            CodeGenerationExpert(self.llm),         # 代码生成专家
            ValidationExpert(self.llm)              # 验证专家
        ]
        
        # 专家执行顺序（简单策略）
        self.expert_order = [0, 1, 2]  # 专家执行顺序索引
    
    def select_expert(self, current_step):
        """选择当前步骤要执行的知识源"""
        if current_step < len(self.expert_order):
            expert_index = self.expert_order[current_step]  # 获取专家索引
            return self.experts[expert_index]  # 返回专家实例
        return None  # 无更多专家
    
    def solve_problem(self, query, requirements):
        """解决问题的主流程"""
        print("=== 黑板架构系统启动 ===")
        
        # 1. 初始化黑板数据
        self.blackboard.write("current_query", query, "controller")  # 写入查询
        self.blackboard.write("requirements", requirements, "controller")  # 写入需求
        self.blackboard.write("solution_status", "initialized", "controller")  # 初始化状态
        
        # 2. 按顺序调用知识源
        current_step = 0  # 当前步骤
        max_steps = len(self.experts)  # 最大步骤数
        
        while current_step < max_steps:
            # 选择当前步骤的专家
            expert = self.select_expert(current_step)  # 选择专家
            
            if expert:
                print(f"\n步骤 {current_step + 1}: 调用{expert.name}")  # 日志输出
                
                # 调用专家贡献
                if expert.name == "语言理解专家":
                    result = expert.contribute(self.blackboard, query)  # 语言理解专家
                else:
                    result = expert.contribute(self.blackboard)  # 其他专家
                
                print(f"{expert.name} 结果: {result['status']}")  # 输出结果
            
            current_step += 1  # 进入下一步
        
        # 3. 获取最终结果
        final_code = self.blackboard.read("generated_code")  # 读取生成代码
        validation = self.blackboard.read("validation_result")  # 读取验证结果
        
        # 4. 返回综合结果
        final_result = {
            "query": query,
            "requirements": requirements,
            "generated_code": final_code,
            "validation": validation,
            "blackboard_state": self.blackboard.get_state(),  # 黑板状态
            "expert_contributions": self.blackboard.read("expert_contributions")  # 专家贡献
        }
        
        print("\n=== 黑板架构系统完成 ===")
        return final_result

# ========== 4. 使用示例 ==========
if __name__ == "__main__":
    # 创建黑板架构系统
    blackboard_system = BlackboardController()  # 创建控制器
    
    # 定义问题和需求
    user_query = "我需要一个LangChain的Agent示例"  # 用户查询
    user_requirements = """
    创建一个简单的LangChain Agent，要求：
    1. 能够回答关于Python编程的问题
    2. 使用ReAct框架
    3. 包含至少两个工具：搜索和计算
    4. 有完整的错误处理
    """  # 用户需求
    
    # 执行黑板架构解决问题
    print(f"问题: {user_query}")
    print(f"需求: {user_requirements}")
    
    result = blackboard_system.solve_problem(user_query, user_requirements)  # 解决问题
    
    # 显示最终结果
    print("\n=== 最终结果 ===")
    print(f"生成的代码长度: {len(result['generated_code']) if result['generated_code'] else 0} 字符")  # 代码长度
    print(f"验证结果: {result['validation'][:100]}..." if result['validation'] else "无验证结果")  # 验证结果
    print(f"专家贡献记录: {result['expert_contributions']}")  # 专家贡献
```

## 三、管道架构(Pipeline Architecture) - 5分钟

### 直观理解
**"流水线作业，每个环节专注一件事"**

### 架构流程
```
输入 → 预处理 → 理解 → 规划 → 执行 → 输出
   ↓       ↓       ↓       ↓       ↓
```

```python
class PipelineAgent:
    def __init__(self):
        # 定义管道中的各个处理阶段
        self.stages = [
            DataPreprocessor(),    # 阶段1：数据预处理
            IntentUnderstanding(), # 阶段2：意图理解  
            TaskPlanner(),         # 阶段3：任务规划
            ActionExecutor()       # 阶段4：动作执行
        ]
    
    def process(self, input_data):
        result = input_data  # 初始化结果为输入数据

        for stage in self.stages:          # 遍历每个阶段

        return result  # 返回最终结果
```

### 3.2 LangChain代码案例（实际实现）
```python
# 管道架构的LangChain实现 - 文本处理流水线示例
from langchain.document_loaders import TextLoader
from langchain.text_splitter import CharacterTextSplitter
from langchain.embeddings import OpenAIEmbeddings
from langchain.vectorstores import Chroma
from langchain.chains import RetrievalQA
from langchain import OpenAI
import os

# 设置API密钥
os.environ["OPENAI_API_KEY"] = "your-api-key-here"

# ========== 1. 管道阶段基类 ==========
class PipelineStage:
    """管道阶段基类"""
    def __init__(self, name):
        self.name = name  # 阶段名称
    
    def process(self, input_data):
        """处理输入数据，需要子类实现"""
        raise NotImplementedError("子类必须实现process方法")  # 抽象方法
    
    def __str__(self):
        return f"PipelineStage({self.name})"  # 字符串表示

# ========== 2. 具体管道阶段实现 ==========
class DocumentLoadingStage(PipelineStage):
    """阶段1：文档加载阶段"""
    def __init__(self):
        super().__init__("文档加载")  # 调用父类初始化
    
    def process(self, file_path):
        """加载文档文件"""
        print(f"[{self.name}] 加载文档: {file_path}")  # 日志输出
        
        try:
            loader = TextLoader(file_path)  # 创建文本加载器
            documents = loader.load()       # 加载文档
            print(f"[{self.name}] 成功加载 {len(documents)} 个文档")  # 成功日志
            return documents  # 返回加载的文档
        except Exception as e:
            print(f"[{self.name}] 加载失败: {str(e)}")  # 错误日志
            return None  # 返回None表示失败

class TextSplittingStage(PipelineStage):
    """阶段2：文本分割阶段"""
    def __init__(self, chunk_size=1000, chunk_overlap=0):
        super().__init__("文本分割")  # 调用父类初始化
        self.chunk_size = chunk_size        # 块大小
        self.chunk_overlap = chunk_overlap  # 重叠大小
    
    def process(self, documents):
        """分割文本为小块"""
        if not documents:
            print(f"[{self.name}] 输入文档为空")  # 空文档警告
            return None  # 返回None
        
        print(f"[{self.name}] 开始分割文本...")  # 开始日志
        
        # 创建文本分割器
        text_splitter = CharacterTextSplitter(
            chunk_size=self.chunk_size,      # 设置块大小
            chunk_overlap=self.chunk_overlap # 设置重叠大小
        )
        
        # 分割文档
        splits = text_splitter.split_documents(documents)  # 分割文档
        print(f"[{self.name}] 分割为 {len(splits)} 个文本块")  # 结果日志
        
        return splits  # 返回分割后的文本块

class EmbeddingStage(PipelineStage):
    """阶段3：向量嵌入阶段"""
    def __init__(self):
        super().__init__("向量嵌入")  # 调用父类初始化
    
    def process(self, text_splits):
        """将文本转换为向量嵌入"""
        if not text_splits:
            print(f"[{self.name}] 输入文本块为空")  # 空输入警告
            return None  # 返回None
        
        print(f"[{self.name}] 开始生成向量嵌入...")  # 开始日志
        
        try:
            # 创建嵌入模型
            embeddings = OpenAIEmbeddings()  # 创建OpenAI嵌入模型
            
            # 创建向量存储
            vectorstore = Chroma.from_documents(
                documents=text_splits,  # 文档列表
                embedding=embeddings,   # 嵌入模型
                persist_directory="./chroma_db"  # 持久化目录
            )
            
            print(f"[{self.name}] 成功创建向量存储")  # 成功日志
            return vectorstore  # 返回向量存储
        except Exception as e:
            print(f"[{self.name}] 嵌入失败: {str(e)}")  # 错误日志
            return None  # 返回None表示失败

class QAChainStage(PipelineStage):
    """阶段4：问答链创建阶段"""
    def __init__(self):
        super().__init__("问答链创建")  # 调用父类初始化
    
    def process(self, vectorstore):
        """创建问答链"""
        if not vectorstore:
            print(f"[{self.name}] 输入向量存储为空")  # 空输入警告
            return None  # 返回None
        
        print(f"[{self.name}] 创建问答链...")  # 开始日志
        
        try:
            # 创建语言模型
            llm = OpenAI(temperature=0)  # 创建OpenAI语言模型，temperature=0保证确定性
            
            # 创建检索问答链
            qa_chain = RetrievalQA.from_chain_type(
                llm=llm,  # 语言模型
                chain_type="stuff",  # 链类型：stuff（简单拼接）
                retriever=vectorstore.as_retriever(),  # 检索器
                return_source_documents=True  # 返回源文档
            )
            
            print(f"[{self.name}] 成功创建问答链")  # 成功日志
            return qa_chain  # 返回问答链
        except Exception as e:
            print(f"[{self.name}] 创建问答链失败: {str(e)}")  # 错误日志
            return None  # 返回None表示失败

class QueryProcessingStage(PipelineStage):
    """阶段5：查询处理阶段"""
    def __init__(self):
        super().__init__("查询处理")  # 调用父类初始化
    
    def process(self, data):
        """处理用户查询"""
        qa_chain = data.get("qa_chain")  # 获取问答链
        query = data.get("query")        # 获取查询
        
        if not qa_chain or not query:
            print(f"[{self.name}] 缺少问答链或查询")  # 参数缺失警告
            return None  # 返回None
        
        print(f"[{self.name}] 处理查询: {query}")  # 开始日志
        
        try:
            # 执行查询
            result = qa_chain({"query": query})  # 执行问答链
            
            print(f"[{self.name}] 查询处理完成")  # 完成日志
            return result  # 返回查询结果
        except Exception as e:
            print(f"[{self.name}] 查询处理失败: {str(e)}")  # 错误日志
            return None  # 返回None表示失败

# ========== 3. 管道控制器 ==========
class PipelineController:
    """管道控制器：管理整个流水线"""
    def __init__(self):
        # 初始化所有管道阶段
        self.stages = [
            DocumentLoadingStage(),      # 阶段1：文档加载
            TextSplittingStage(),        # 阶段2：文本分割
            EmbeddingStage(),            # 阶段3：向量嵌入
            QAChainStage(),              # 阶段4：问答链创建
            QueryProcessingStage()       # 阶段5：查询处理
        ]
        
        # 中间数据存储
        self.intermediate_data = {}  # 存储中间结果
    
    def execute_pipeline(self, file_path, query):
        """执行完整管道流程"""
        print("=== LangChain管道架构系统启动 ===")
        
        current_data = {"file_path": file_path, "query": query}  # 初始数据
        
        # 依次执行每个阶段
        for i, stage in enumerate(self.stages):
            stage_num = i + 1  # 阶段编号
            print(f"\n[阶段 {stage_num}/{len(self.stages)}] {stage.name}")  # 阶段开始
            
            # 执行当前阶段
            result = stage.process(current_data)  # 处理当前数据
            
            if result is None:
                print(f"[错误] 阶段 {stage_num} 执行失败，管道终止")  # 错误日志
                return None  # 终止管道
            
            # 更新当前数据
            if stage.name == "文档加载":
                current_data["documents"] = result  # 存储文档
            elif stage.name == "文本分割":
                current_data["text_splits"] = result  # 存储文本块
            elif stage.name == "向量嵌入":
                current_data["vectorstore"] = result  # 存储向量存储
            elif stage.name == "问答链创建":
                current_data["qa_chain"] = result  # 存储问答链
            elif stage.name == "查询处理":
                current_data["final_result"] = result  # 存储最终结果
            
            # 保存中间数据（用于调试）
            self.intermediate_data[stage.name] = {
                "stage_num": stage_num,
                "result_type": type(result).__name__,
                "success": True
            }
        
        print("\n=== LangChain管道架构系统完成 ===")
        return current_data  # 返回最终数据

# ========== 4. 使用示例 ==========
if __name__ == "__main__":
    # 创建管道系统
    pipeline_system = PipelineController()  # 创建管道控制器
    
    # 定义输入
    document_file = "example.txt"  # 文档文件路径
    user_query = "LangChain的主要特点是什么？"  # 用户查询
    
    print(f"文档文件: {document_file}")  # 显示文档文件
    print(f"用户查询: {user_query}")     # 显示用户查询
    
    # 执行管道流程
    final_result = pipeline_system.execute_pipeline(document_file, user_query)  # 执行管道
    
    # 显示结果
    if final_result and "final_result" in final_result:
        result = final_result["final_result"]  # 获取最终结果
        
        print("\n=== 查询结果 ===")
        print(f"答案: {result['result']}")  # 显示答案
        
        if result.get('source_documents'):
            print(f"\n参考文档:")  # 显示参考文档
            for i, doc in enumerate(result['source_documents'][:3]):  # 只显示前3个
                print(f"  {i+1}. {doc.page_content[:100]}...")  # 显示文档内容前100字符
    else:
        print("\n管道执行失败")  # 失败提示
    
    # 显示管道执行统计
    print("\n=== 管道执行统计 ===")
    for stage_name, info in pipeline_system.intermediate_data.items():
        print(f"{stage_name}: 阶段{info['stage_num']}, 结果类型: {info['result_type']}")  # 显示阶段信息
```
